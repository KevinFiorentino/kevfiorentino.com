export async function getStaticPaths() {

  const pages: any[] = [];

  // Build pages of each content
  for await (let c of collectionMetadata) {
    const coll = await getCollection(c.name as any);
    coll.map((post) => {
      pages.push({
        params: {
          lang: post.slug.split('/')[0],
          category: c.name,
          slug: post.slug.slice(3),
        },
        props: {
          category: c.name,
          post: post,
        },
      });
    });
  }

  return pages;
}
